# Pathfinder TSP — v13
# requirements.txt
#
# This is a Node.js / React project. Dependencies are managed by npm (package.json).
# This file documents every dependency with its pinned version and purpose
# for human reference, onboarding, and audit.
#
# Install all: npm install
# ─────────────────────────────────────────────────────────────────────────────


# ── Runtime dependencies ──────────────────────────────────────────────────

react@18.3.1
# UI component framework. Functional components + hooks throughout.

react-dom@18.3.1
# React renderer for the browser DOM.

react-router-dom@6.26.1
# Client-side routing. Routes: / (Landing), /arena, /results, /global.
# vercel.json rewrites all paths to index.html for SPA support.

zustand@4.5.4
# Lightweight global state management.
# Three stores: gameStore (nodes, edges, scores), aiStore (AI state), uiStore (theme, view).

pixi.js@7.4.2
# 2D WebGL canvas renderer. Used for all 2D node/edge drawing and mouse/touch interaction.
# Sprites, graphics, and interactive hit areas for the game canvas.

three@0.164.1
# 3D WebGL renderer. Used for the optional 3D view (toggle in ArenaScreen).
# OrbitControls for drag-to-orbit, raycasting for node click detection.

gsap@3.12.5
# Animation library. Used for node spawn animations, edge draw animations,
# and completion transitions in the canvas hooks.

recharts@2.12.7
# React charting library. Used in:
#   - CollaborationScore.jsx  (donut pie chart — human vs AI edge contribution)
#   - ConvergenceGraph.jsx    (line chart — path length convergence over rounds)

# ── Dev dependencies ──────────────────────────────────────────────────────

vite@5.4.1
# Build tool and dev server. ES module bundling, HMR, and optimised production builds.

@vitejs/plugin-react@4.3.1
# Vite plugin for React — enables JSX transform and fast refresh.

tailwindcss@3.4.10
# Utility-first CSS framework. All layout, typography, and colour classes are Tailwind.
# Custom design tokens (game-bg, game-cyan, game-amber, etc.) defined in tailwind.config.js.

postcss@8.4.41
# CSS post-processor. Required by Tailwind for class generation.

autoprefixer@10.4.20
# PostCSS plugin that adds vendor prefixes automatically.

@types/react@18.3.3
# TypeScript type definitions for React (used by editor tooling even without TS compilation).

@types/react-dom@18.3.0
# TypeScript type definitions for ReactDOM.


# ── Node / runtime requirements ───────────────────────────────────────────

# Node.js  >= 18.0.0   (ES2022 syntax, native fetch, structuredClone)
# npm      >= 9.0.0    (workspaces-compatible, lockfile v3)
# Browser  Chromium 110+ / Firefox 115+ / Safari 16.4+
#          (WebGL 2.0 required for PixiJS and Three.js)


# ── No backend dependencies ───────────────────────────────────────────────
# This is a fully static front-end app.
# All data is in-memory per session.
# Leaderboard and convergence data are mocked in src/utils/mockAI.js.
# Deployment target: Vercel (static, zero config beyond vercel.json).
